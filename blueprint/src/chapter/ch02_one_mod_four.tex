\chapter{Roadmap: The $d \equiv 1 \pmod{4}$ Branch}

This chapter outlines the planned formalization of the remaining branch
of the ring-of-integers classification.  When $d \equiv 1\!\pmod{4}$,
the ring of integers is $\Z\!\left[\frac{1+\sqrt{d}}{2}\right]$,
which is strictly larger than $\Z[\sqrt{d}]$.

\begin{definition}[Half-order generator]\label{def:plan_half_order}
  Define $\omega_d := \frac{1+\sqrt{d}}{2} = \texttt{halfInt}(d,1,1)$
  in $\Q(\sqrt{d})$ and formalize its basic arithmetic:
  $\omega_d^2 = \omega_d + \frac{d-1}{4}$.
  \notready
  \uses{def:Qsqrtd, def:halfInt}
\end{definition}

\begin{definition}[Half-order ring]\label{def:plan_half_order_ring}
  Define $\texttt{ZQuadHalf}(d)$ as a $\Z$-module with basis
  $\{1, \omega_d\}$ and the induced multiplication, providing an
  analogue of $\texttt{ZQuad}(d)$ for the $d \equiv 1\!\pmod{4}$ case.
  \notready
  \uses{def:plan_half_order, def:ZQuad}
\end{definition}

\begin{theorem}[Same-parity membership criterion]\label{thm:plan_same_parity_membership}
  For $d \equiv 1\!\pmod{4}$, the element $\texttt{halfInt}(d,a',b')$ lies
  in the image of $\texttt{ZQuadHalf}(d) \hookrightarrow \Q(\sqrt{d})$
  if and only if $a' \equiv b' \pmod{2}$.
  \notready
  \uses{cor:mod4_same_parity, def:plan_half_order_ring}
\end{theorem}

\begin{theorem}[Integrality $\Rightarrow$ half-order representability]\label{thm:plan_exists_zquadhalf_of_isIntegral}
  For squarefree $d \equiv 1\!\pmod{4}$, every integral element of
  $\Q(\sqrt{d})$ is in the image of $\texttt{ZQuadHalf}(d)$.

  The proof should follow the same strategy as
  \cref{thm:exists_zquad_of_isIntegral}: extract integer trace and norm,
  represent as $\texttt{halfInt}$, then use the same-parity criterion
  from \cref{cor:mod4_same_parity} instead of the even-even criterion.
  \notready
  \uses{thm:plan_same_parity_membership, thm:trace_halfInt, thm:norm_halfInt}
\end{theorem}

\begin{theorem}[$\mathcal{O}_{\Q(\sqrt{d})} \cong \Z[\omega_d]$ for $d \equiv 1\pmod{4}$]\label{thm:plan_one_mod_four_branch}
  For squarefree $d \equiv 1\!\pmod{4}$:
  \[
    \mathcal{O}_{\Q(\sqrt{d})} \;\cong\; \Z\!\left[\tfrac{1+\sqrt{d}}{2}\right].
  \]
  \notready
  \uses{thm:plan_exists_zquadhalf_of_isIntegral, def:plan_half_order_ring}
\end{theorem}
