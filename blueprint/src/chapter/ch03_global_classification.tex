\chapter{Roadmap: Global Ring-of-Integers Classification}

\begin{theorem}[Unified classification of $\mathcal{O}_{\Q(\sqrt{d})}$]\label{thm:plan_global_classification}
  For every squarefree $d \neq 0,1$:
  \[
    \mathcal{O}_{\Q(\sqrt{d})} \;\cong\;
    \begin{cases}
      \Z[\sqrt{d}] & \text{if } d \not\equiv 1\!\pmod{4}, \\[4pt]
      \Z\!\left[\dfrac{1+\sqrt{d}}{2}\right] & \text{if } d \equiv 1\!\pmod{4}.
    \end{cases}
  \]
  This combines the two branches from
  \cref{thm:ringOfIntegers_equiv_zquad} and
  \cref{thm:plan_one_mod_four_branch}.
  \notready
  \uses{thm:ringOfIntegers_equiv_zquad, thm:plan_one_mod_four_branch}
\end{theorem}

\begin{corollary}[Explicit integral basis]\label{cor:plan_explicit_basis}
  Extract from the unified classification a practical API: an explicit
  $\Z$-basis $\{1, \omega\}$ for $\mathcal{O}_{\Q(\sqrt{d})}$
  (where $\omega = \sqrt{d}$ or $\frac{1+\sqrt{d}}{2}$) together
  with a decidable membership predicate.
  \notready
  \uses{thm:plan_global_classification}
\end{corollary}
